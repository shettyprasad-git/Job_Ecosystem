Docker Compose ‚Äî One Command to Rule Them All

docker-compose.yml
docker compose up -d 
# Run everything: docker compose up -d
services:

  db:                              # PostgreSQL
    image: postgres:16-alpine
    environment:
      POSTGRES_DB: jobeco
      POSTGRES_USER: dev
      POSTGRES_PASSWORD: devpass
    volumes: ["postgres_data:/var/lib/postgresql/data"]
    ports: ["5432:5432"]

  redis:                           # Redis (message broker)
    image: redis:7-alpine
    ports: ["6379:6379"]

  backend:                         # FastAPI on port 8000
    build: ./backend
    ports: ["8000:8000"]
    volumes: ["./backend:/app"]  # live reload
    depends_on: [db, redis, ollama]
    env_file: .env.local

  frontend:                        # React Vite on port 5173
    build: ./frontend
    ports: ["5173:5173"]
    volumes: ["./frontend:/app"]  # live reload

  celery:                          # Async task worker
    build: ./backend
    command: celery -A app.tasks worker --loglevel=info
    depends_on: [db, redis, ollama]

  celery_beat:                     # Scheduled scraper cron
    build: ./backend
    command: celery -A app.tasks beat --loglevel=info

  flower:                          # Task monitor UI ‚Äî localhost:5555
    image: mher/flower
    ports: ["5555:5555"]

  pgadmin:                         # Database UI ‚Äî localhost:5050
    image: dpage/pgadmin4
    ports: ["5050:80"]
    environment:
      PGADMIN_DEFAULT_EMAIL: admin@dev.com
      PGADMIN_DEFAULT_PASSWORD: admin

  mailhog:                         # Email catcher ‚Äî localhost:8025
    image: mailhog/mailhog
    ports: ["8025:8025", "1025:1025"]

  ollama:                          # Local LLM ‚Äî localhost:11434
    image: ollama/ollama
    ports: ["11434:11434"]
    volumes: ["ollama_models:/root/.ollama"]
    # Pull model after first run:
    # docker exec -it ollama ollama pull llama3.1

Quick Start Commands

bash ‚Äî get running in 10 minutes
# 1. Prerequisites ‚Äî install these first
brew install docker           # or download Docker Desktop
brew install node git         # Node 20+, Git
brew install python@3.12      # Python 3.12

# 2. Clone & scaffold your project
git init job-ecosystem && cd job-ecosystem

# 3. Spin up all infrastructure
docker compose up -d

# 4. Pull your local LLM (one-time download)
docker exec -it job-ecosystem-ollama-1 ollama pull llama3.1

# 5. Bootstrap the database
docker exec backend alembic upgrade head

# 6. Install Python deps
cd backend && pip install fastapi sqlalchemy celery redis scrapy playwright

# 7. Install frontend deps
cd frontend && npm install && npm run dev

# 8. Open your browser
# App:      http://localhost:5173
# API docs: http://localhost:8000/docs
# Emails:   http://localhost:8025
# DB:       http://localhost:5050
# Tasks:    http://localhost:5555
